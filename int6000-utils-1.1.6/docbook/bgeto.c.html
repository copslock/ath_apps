#ifndef BGETO_SOURCE
#define BGETO_SOURCE

#include <errno.h>
#include <stdint.h>
#include <stdlib.h>

#include "../tools/buffer.h"

size_t bgeto (buffer_t *b) 

{
	if (b == 0) 
	{
		errno = EINVAL;
		return 0;
	}
	return (b->bcurr - b->bstart);
}

#endif
