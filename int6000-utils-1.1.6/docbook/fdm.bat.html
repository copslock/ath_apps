:ONE

	int6k -i %ETH% -p %PIB% %DEV% 
	psin %PIB% < %DIR%/INT6400-Wall-NA.txt
	setpib %PIB% a0c byte 0
	int6k -i %ETH% -P %PIB% %DEV%
	int6k -i %ETH% -C2 %DEV%

	GOTO :END
