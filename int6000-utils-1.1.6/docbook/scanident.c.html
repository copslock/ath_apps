#ifndef SCANIDENT_SOURCE
#define SCANIDENT_SOURCE

#include <ctype.h>

#include "../scan/scan.h"

void scanident (SCAN * content) 

{
	while ((content->final < content->limit && (isalnum (*content->final) || (*content->final == '-')))) 
	{
		content->final++;
	}
	return;
}

#endif
