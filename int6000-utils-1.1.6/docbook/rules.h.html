#ifndef __GNUC__
#pragma pack (push,1)
#endif

typedef struct __packed classifier 

{
	uint8_t CR_ID;
	uint8_t CR_OPERAND;
	uint8_t CR_VALUE [16];
}

classifier;
typedef struct __packed cspec 

{
	uint16_t CSPEC_VERSION;
	uint32_t VLAN_TAG;
	uint32_t RSVD;
}

cspec;
typedef struct __packed rule 

{
	uint8_t MCONTROL;
	uint8_t MVOLATILITY;
	uint8_t MACTION;
	uint8_t MOPERAND;
	uint8_t NUM_CLASSIFIERS;
	struct classifier CLASSIFIER [3];
	struct cspec cspec;
}

rule;
