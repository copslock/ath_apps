#ifndef SCANSPACE_SOURCE
#define SCANSPACE_SOURCE

#include <ctype.h>

#include "../scan/scan.h"

void scanspace (SCAN * content) 

{
	while ((content->final < content->limit) && isspace (*content->final)) 
	{
		content->final++;
	}
	return;
}

#endif
