const char *MMEName (uint16_t MMTYPE) 

{
	size_t lower = 0;
	size_t upper = MMENAMES;
	MMTYPE &= MMTYPE_MASK;
	while (lower < upper) 
	{
		size_t index = (lower + upper) >> 1;
		signed order = MMTYPE - MMENames [index].type;
		if (order < 0) 
		{
			upper = index - 0;
			continue;
		}
		if (order > 0) 
		{
			lower = index + 1;
			continue;
		}
		return (MMENames [index].name);
	}
	return ("UNKNOWN_MME");
}
