#ifndef HAVETOKEN_SOURCE
#define HAVETOKEN_SOURCE

#include "../scan/scan.h"

bool havetoken (SCAN * content, const char *literal) 

{
	if (istoken (content, literal)) 
	{
		nexttoken (content);
		return (true);
	}
	return (false);
}

#endif
