#ifndef SEQUENCETERMINATE_SOURCE
#define SEQUENCETERMINATE_SOURCE

#include <string.h>

#include "ihpapi.h"
#include "ihp.h"
 
void ihpapi_SequenceTerminate () 

{
	extern struct SeqCB scb;
	bclose (scb.bcb [0]);
	bclose (scb.bcb [1]);
	memset (&scb, 0, sizeof (scb));
	return;
}

#endif
