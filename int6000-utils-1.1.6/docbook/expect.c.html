#include <stdio.h>

#include "../tools/symbol.h"

void expect (const struct _code_ list [], size_t size, FILE *fp) 

{
	const struct _code_ * item = list;
	if (list) while ((size_t)(item - list) < size) 
	{
		if (item > list) 
		{
			fprintf (fp, "|");
		}
		fprintf (fp, "'%s'", item->name);
		item++;
	}
	return;
}
