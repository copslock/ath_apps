#ifndef REORDER_SOURCE
#define REORDER_SOURCE

#include "../nodes/nodes.h"

#if NODE_LIST

void reorder (NODE * node) 

{
	static unsigned index = 0;
	static unsigned level = 0;
	while (node) 
	{
		node->index = index++;
		node->level = level;
		node->count = index;
		level++;
		reorder (node->below);
		level--;
		node->count = index - node->count;
		node = node->after;
	}
	return;
}

#endif
