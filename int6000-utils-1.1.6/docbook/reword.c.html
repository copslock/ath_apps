#ifndef REWORD_SOURCE
#define REWORD_SOURCE

#include <unistd.h>
#include <string.h>

#include "../tools/symbol.h"

const char * reword (code_t code, const struct _code_ list [], size_t size) 

{
	const struct _code_ * item = list;
	while ((unsigned)(item - list) < size) 
	{
		if (item->code == code) 
		{
			return (item->name);
		}
		item++;
	}
	return (NULL);
}

#endif
