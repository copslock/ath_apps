#ifndef XMLCONTENT_SOURCE
#define XMLCONTENT_SOURCE

#include <string.h>

#include "../nodes/nodes.h"

const char * xmlcontent (NODE * node) 

{
	if (node) 
	{
		node = node->below;
	}
	while (node) 
	{
		if (node->type == NODE_DATA) 
		{
			return (node->text);
		}
		node = node->after;
	}
	return ("");
}

#endif
