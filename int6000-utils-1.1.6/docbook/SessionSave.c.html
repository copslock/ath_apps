#ifndef SESSIONSAVE_SOURCE
#define SESSIONSAVE_SOURCE

#include <unistd.h>
#include <fcntl.h>

#include "../cgi/session.h"

void SessionSave (const char * filename) 

{
	extern struct session session;
	signed fd = open (filename, O_CREAT|O_TRUNC|O_WRONLY);
	write (fd, &session, sizeof (session));
	close (fd);
	return;
}

#endif
