#ifndef MMEMODE_SOURCE
#define MMEMODE_SOURCE

#include <stdint.h>

#include "../int6k/homeplug.h"

const char *MMEMode (uint16_t MMTYPE) 

{
	static const char * MMEMODE [] = 
	{
		"REQ",
		"CNF",
		"IND",
		"RSP" 
	};
	return (MMEMODE [MMTYPE & MMTYPE_MODE]);
}
