#ifndef BOPEN_SOURCE
#define BOPEN_SOURCE
#include <errno.h>
#include <stdint.h>
#include <stdlib.h>

#include "../tools/buffer.h"

buffer_t * bopen (uint8_t b [], size_t blen, size_t bseglen) 

{
	buffer_t * buffer;
	if (b == 0) 
	{
		errno = EINVAL;
		return (0);
	}
	if (blen == 0 || bseglen == 0) 
	{
		errno = EINVAL;
		return (0);
	}
	if ((buffer = (buffer_t *)malloc (sizeof (buffer_t))) == 0) 
	{
		return (0);
	}
	buffer->bstart = buffer->bcurr = b;
	buffer->bend = b + blen;
	buffer->blen = blen;
	buffer->bseglen = bseglen;
	buffer->bcurrsegnum = 0;
	return buffer;
}

#endif
